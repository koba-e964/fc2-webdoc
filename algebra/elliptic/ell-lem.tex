\section{レムニスケートサイン/コサイン函数}
\subsection{実函数としてのレムニスケートサイン函数}
\begin{defn}
レムニスケートサイン函数(lemniscate sine function)$\slem(u) $を、以下の関数$u(s)\ (0\leq s\leq 1)$の逆関数として定義する.
\[
u(s)=\int_0^r \frac 1 {\sqrt{1-r^4}} dr
\]
また,レムニスケート周率(lemniscate constant)を,
\[
\varpi:=2u(1)=2\int_0^1 \frac 1 {\sqrt{1-r^4}} dr
\]
で定める.(約2.6221)
\end{defn}
$\slem(u)$は$0\leq u \leq \varpi/2$で定義された関数である.また$\slem(u)$は以下の性質を持っている.(証明略)
\begin{prop}$\slem(u) (0\leq u \leq \varpi/2)$は以下を満たす.
\[\slem(0)=0\]
\begin{equation}\slem'(u)=\sqrt{1-\slem(u)^4}\label{1}\end{equation}
\[\slem\left(\frac \varpi 2\right)=1\]
\[\slem(u+v)=\frac{\slem(u)(1+\slem(u)^2)\sqrt{1-\slem(v)^4}+\slem(v)(1+\slem(v)^2)\sqrt{1-\slem(u)^4}}
{(1+\slem(u)^2)(1+\slem(v)^2)-\slem(u)\slem(v)\sqrt{1-\slem(u)^4}\sqrt{1-\slem(v)^4}}
\quad\mbox{(加法公式)}\]
\end{prop}
$\slem(u)$の定義域を以下のようにして$0\leq u \leq 2\varpi$に拡張することができる.
\begin{align*}
\left\{ 
\begin{array}{l l}
\slem(u)=\slem(\varpi-u) &\quad\mbox{(if }\frac\varpi 2\leq u \leq \varpi) \\
\slem(u)=-\slem(u-\varpi) &\quad\mbox{(if }\varpi\leq u \leq 2\varpi)
\end{array}
\right.
\end{align*}
またこのようにして拡張した$\slem(u)$を周期$2\varpi$の周期函数と見ることによって定義域を$\R$全体に拡張することができる.
以降$\slem(u)$はこのようにして定義域を$\R$全体に拡張した函数のことを指すものとする.
\begin{prop}
函数$\slem:\R\rightarrow\R$は連続で,微分可能であり,周期$2\varpi$である.
またこれに対しても加法定理は成り立つ($\sqrt{1-\slem(u)^4}$などに適切に符号をつければ).
\end{prop}
\begin{rema}
このとき(\ref{1})式は必ずしも成り立たない($\slem'(u)$が負になることもある)ことに注意すること.

なお(\ref{1})式の両辺を2乗した
\[
\slem'(u)^2=1-\slem(u)^4
\]
は$\R$全体で成り立つ.
\end{rema}
このままでは加法定理に根号が入っており$0\leq u \leq \varpi$の場合を扱うときに困るので,これを消去することを考える.
そのために函数$\clem(u)$を定義する.

加法公式に$v=\varpi/2$を代入する.($0\leq u \leq \varpi/2$)
\begin{align*}
\slem\left(u+\frac{\varpi}{2}\right)&=
\frac{\slem(u)(1+\slem(u)^2)\sqrt{1-1^4}+1(1+1^2)\sqrt{1-\slem(u)^4}}
{(1+\slem(u)^2)(1+1^2)-\slem(u)\cdot 1 \cdot\sqrt{1-\slem(u)^4}\sqrt{1-1^4}} \\
&=\frac{\sqrt{1-\slem(u)^4}}{1+\slem(u)^2}
\end{align*}
$0\leq u \leq \varpi/2$かつ$v=\varpi/2$であるため根号の中身は負にはならない.
これをレムニスケートコサイン函数(lemniscate cosine function)と呼ぶことにする.
\begin{defn}
レムニスケートコサイン函数(lemniscate cosine function) $\clem(u)\ (u\in \R)$を以下のように定義する.
\[
\clem(u):=\slem\left(u+\frac \varpi 2\right)
\]
また,$0\leq u \leq \varpi/2$とすると
\[
\clem(u)=\frac{\sqrt{1-\slem(u)^4}}{1+\slem(u)^2}
\]
が成立する.
\end{defn}
これを用いると加法公式は($0\leq u,v\leq\varpi/2$の範囲では)以下のように書き直せる.
\begin{theo}(加法定理)\\
$0\leq u,v\leq\varpi/2$のとき,以下が成立する.またこの式は一般に$u,v\in \R$のとき成立する.
\[
\slem(u+v)=\frac{\slem(u)\clem(v)+\clem(u)\slem(v)}{1-\slem(u)\slem(v)\clem(u)\clem(v)}
\]
\end{theo}
以上により,周期$2\varpi$の周期函数$\slem(u),\clem(u)$が定義された.

レムニスケートサイン/コサイン函数は以下の性質を持つ.
\begin{prop}以下が成立する.
\begin{enumerate}
\item $\slem,\clem$の特殊値
\[\slem(0)=0,\ \clem(0)=1,\ \slem(\varpi/2)=1,\ \clem(\varpi/2)=0\]
\item 導関数
\[
\slem'(u)=\clem(u)(1+\slem(u)^2), \clem'(u)=-\slem(u)(1+\clem(u)^2)
\]
\item 両者の間の恒等式
\[
(1+\slem(u)^2)(1+\clem(u)^2)=2\quad(\forall u\in\R)
\]
\item 両者の間の恒等式(2)
\[
\slem\left(u+\frac\varpi 2\right)=\clem(u),\ \clem\left(u+\frac\varpi 2\right)=-\slem(u)
\]
\item 対称性
\[
\slem(-u)=-\slem(u),\ \clem(-u)=\clem(u)
\]
\item 加法定理
\begin{align*}
\slem(u+v)&=\frac{\slem(u)\clem(v)+\clem(u)\slem(v)}{1-\slem(u)\slem(v)\clem(u)\clem(v)}\\
\clem(u+v)&=\frac{\clem(u)\clem(v)-\slem(u)\slem(v)}{1+\slem(u)\slem(v)\clem(u)\clem(v)}
\end{align*}
\item 2倍角の公式
\[
\slem(2u)=\frac{2\slem(u)\clem(u)(1+\slem(u)^2)}{1+\slem(u)^4},\ 
\clem(2u)=\frac{1-2\slem(u)^2-\slem(u)^4}{1+2\slem(u)^2-\slem(u)^4}
=-\frac{1-2\clem(u)^2-\clem(u)^4}{1+2\clem(u)^2-\clem(u)^4}
\]
\end{enumerate}
\end{prop}
\subsection{複素函数としてのレムニスケートサイン/コサイン}
\subsubsection*{$\slem(u)$のTaylor展開}
$\slem'(u)$は$\slem(u)$と$\clem(u)$で表せるので,$\slem(u)$は無限回微分可能である.
よって$z=0$の周りにおけるTaylor展開が存在する.関係式
\[
\slem'(u)=\sqrt{1-\slem(u)^4}
\]
を用いて,
\begin{align*}
\slem(u)=u+O(u^2)\longrightarrow\slem'(u)=\sqrt{1-(u^4+O(u^5))}=1-\frac 12 u^4+O(u^5)\longrightarrow \slem(u)=u-\frac 1{10}u^5+O(u^6)\\
\longrightarrow \slem'(u)=\sqrt{1-\left(u^4-\frac 25 u^8+O(u^9)\right)}=1-\frac 12 u^4+\frac 3{40}u^8+O(u^9)\longrightarrow \slem(u)=u-\frac 1{10} u^5 +\frac 1{120}u^9+O(u^{10})
\end{align*}
を得る.ここで,以下の命題が成り立つ.
\begin{prop}
$\slem(u)$のTaylor展開を
\[
\slem(u)=\sum_{n=0}^\infty b_nu^n
\]
とする.このとき,
\[
b_n=0 \quad\mbox{(if  } n \not \equiv 1 \pmod 4)
\]
である.(証明略)
\end{prop}
この事実から,$\slem(u)$のTaylor展開に$u=it$を形式的に代入することで,
\[
\slem(it)=i\slem(t)
\]
が得られる.またこのとき,
\[
\clem(it)=\frac{\sqrt{1-\slem(it)^4}}{1+\slem(it)^2}=\frac{\sqrt{1-\slem(t)^4}}{1-\slem(t)^2}=\frac 1 {\clem(t)}
\]
も導かれる.この式を元にして$\slem(u),\clem(u)$の定義域を複素数全体に拡張する.
\begin{defn}
$u,v\in \R$について,
\[\slem(it)=i\slem(t),\ \clem(it)=\frac 1{\clem(t)}\]と定め,
$u+iv\in\C$については,加法定理を形式的に用いて
\begin{align*}
\slem(u+iv)&:=\frac{\slem(u)\clem(iv)+\clem(u)\slem(iv)}{1-\slem(u)\slem(iv)\clem(u)\clem(iv)} \\
&=\frac{\slem(u)/\clem(v)+i\clem(u)\slem(v)}{1-i\slem(u)\slem(v)\clem(u)/\clem(v)}
\end{align*}
\begin{align*}
\clem(u+iv)&:=\frac{\clem(u)\clem(iv)-\slem(u)\slem(iv)}{1+\slem(u)\slem(iv)\clem(u)\clem(iv)} \\
&=\frac{\clem(u)/\clem(v)-i\slem(u)\slem(v)}{1+i\slem(u)\slem(v)\clem(u)/\clem(v)} 
\end{align*}
と定める.このとき$\slem,\clem:\C\rightarrow\C\cup\{\infty\}$は(一部の点を除いて定義される)複素函数であり,
加法定理と
\[\slem(iz)=i\slem(z),\ \clem(iz)=\frac 1{\clem(z)}\quad(\forall z\in\C)\]
および関係式
\begin{equation}
(1+\slem(z)^2)(1+\clem(z)^2)=2\quad(\forall z\in\C) \label{2}
\end{equation}
を満たす.(証明略)
\end{defn}
\begin{rema}
$\slem,\clem$は($\C\rightarrow\C$の写像としてみたとき)定義されない点を持つ.その場合は$\slem(z)=\infty$と定める.
\end{rema}
以降は引数が複素数であることを強調するために引数として文字$z$を使う.

$\slem(z),\clem(z)$の特殊値を求める.
\begin{prop}
\[
\slem\left(\frac 12 \varpi i\right)=i,\ \slem\left(\frac 12 \varpi(1+i)\right)=\infty,
\ \clem\left(\frac 12 \varpi i\right)=\infty,\ \clem\left(\frac 12 \varpi(1+i)\right)=-i
\]
\end{prop}
(証明) 
\[
\slem\left(\frac 12 \varpi i\right)=i\slem\left(\frac 12 \varpi\right)=i\cdot 1=i
\]
\[
\clem\left(\frac 12 \varpi i\right)=\frac 1{\clem\left(\frac 12 \varpi\right)}=\frac 10=\infty
\]
\[
\slem\left(\frac 12 \varpi(1+i)\right)=\clem\left(\frac 12 \varpi i\right)=\infty
\]
\[
\clem\left(\frac 12 \varpi(1+i)\right)=-\slem\left(\frac 12 \varpi i\right)=-i
\]\Qed

\begin{prop}
\[\slem(\varpi(1+i))=0,\ \clem(\varpi(1+i))=1\]
\end{prop}
(証明)\\
2倍角の公式を用いる.$c:=\clem(\varpi(1+i)/2)=-i$とおく.
\[
\clem(\varpi(1+i))=-\frac {1-2c^2-c^4}{1+2c^2-c^4}=-\frac{1+2-1}{1-2-1}=1
\]
また,関係式(\ref{2})
\[
(1+\slem(z)^2)(1+\clem(z)^2)=2\quad(\forall z\in\C)
\]より
\[
(1+\slem(\varpi(1+i))^2)(1+\clem(\varpi(1+i))^2)=2
\]
\[
(1+\slem(\varpi(1+i))^2)(1+1)^2)=2
\]
\[
1+\slem(\varpi(1+i))^2=1
\]
\[
\slem(\varpi(1+i))=0
\]
\Qed
以上より,$z=\varpi(1+i)$での$\slem,\clem$の値は$z=0$での値に等しい.
また,$\slem,\clem$の導関数はすべて$\slem,\clem$の式で表せるので,$z=\varpi(1+i)$での$\slem,\clem$の任意の階数の微分の値は$z=0$での値に等しい.
よって$z=\varpi(1+i)$近傍の振る舞いと$z=0$近傍の振る舞いは同じであり,$\slem,\clem$は周期$\varpi(1+i)$の周期函数である.
これを直接的に示す.
\begin{theo}
\[
\slem(z+\varpi(1+i))=\slem(z),
\ \clem(z+\varpi(1+i))=\clem(z)
\]
\end{theo}
(証明)加法定理より明らか.\Qed
以上より$\slem(z),\clem(z)$は{\bf 二重周期函数}である.
